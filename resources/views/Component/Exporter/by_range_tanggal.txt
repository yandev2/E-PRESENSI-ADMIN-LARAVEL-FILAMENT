use Carbon\Carbon;

$tanggalAkhir = Carbon::parse($json['tanggal_akhir']);

$tanggalList = [];
$mingguList = [];

$current = $tanggalAwal->copy();

while ($current->lte($tanggalAkhir)) {
$tanggalList[] = $current->copy();

if ($current->isSunday()) {
$mingguList[] = $current->format('Y-m-d');
}

}

$jumlahHari = count($tanggalList);
