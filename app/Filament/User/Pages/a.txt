class Dashboard extends BaseDashboard implements HasInfolists
{
    use HasPageShield, InteractsWithInfolists;

   protected string $view = 'filament.user.pages.dashboard';

    public function getTitle(): string|Htmlable
    {

        return 'Dashboards';
    }
    protected function infolist(Schema $schema): Schema
    {
        return $schema
            ->components([
                Section::make()
                    ->extraAttributes(['class' => 'account_dashboard'])
                    ->columns([
                        'sm' => 3,
                        'md' => 3,
                        'lg' => 3,
                        'xl' => 3,
                        '2xl' => 8
                    ])
                    ->schema([
                        Grid::make(1)
                            ->columnSpan(7)
                            ->schema([
                                TextEntry::make('halo')
                                    ->hiddenLabel()
                                    ->alignLeft()
                                    ->state('Halo, Selamat Datang Kembali 👋​')
                                    ->alignment(Alignment::Left)
                                    ->weight(FontWeight::Bold)
                                    ->size(TextSize::Large)
                                    ->belowContent(
                                        Text::make(fn() => auth()->user()->name)
                                            ->weight(FontWeight::Light)
                                            ->size(TextSize::Small)
                                    )
                            ]),
                        Actions::make([
                            Action::make('logout')
                                ->label('Logout')
                                ->button()
                                ->icon(Heroicon::ArrowLeftEndOnRectangle)
                                ->color('danger')
                                ->requiresConfirmation()
                                ->modalHeading('KONFIRMASI')
                                ->modalWidth(Width::Small)
                                ->modalDescription('konfirmasi untuk melakukan logout')
                                ->action(function () {
                                    Auth::logout();
                                    request()->session()->invalidate();
                                    request()->session()->regenerateToken();

                                    return redirect(Filament::getLoginUrl());
                                })

                        ])
                            ->columnSpan(1)
                            ->alignEnd()
                    ]),
            ]);
    }
    public array $filter = [];

    public function content(Schema $schema): Schema
    {
        return $schema
            ->components([
                ...(method_exists($this, 'getFiltersForm') ? [$this->getFiltersFormContentComponent()] : []),
                $this->getWidgetsContentComponent(),
            ]);
    }

    protected function widgetCustom(Schema $schema): Schema
    {
        return $schema
            ->components([

              
                Grid::make()
                    ->columns([
                        "sm" => 1,
                        "md" => 3,
                        "lg" => 3,
                        "xl" => 3,
                        "2xl" => 3
                    ])
                    ->schema([
                        Livewire::make(AnalyticKehadiranMasuk::class, [
                            'filters' => fn() => ['sds'],
                        ]),
                        Livewire::make(AnalyticKehadiranKeluar::class),
                        Livewire::make(AnalyticKetidakHadiran::class),
                    ]),
              //  Livewire::make(AnalyticKehadiranBulanan::class)->extraAttributes(['class' => 'chart-kehadiran-bulanan']),
            ]);
    }
}
